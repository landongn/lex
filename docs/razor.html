copies one object to another, optionally performing some kind of transform.

 
      copy: function(target, source, overwrite, transform) {
        for (var key in source) {
          if (overwrite || typeof target[key] === 'undefined') {
              target[key] = transform ? transform(source[key]) :  source[key];
          }
        }
        return target;
      },

creates a namespaced object, forcing it to exist within the razor namespace. don't use this function directly, use .extend(namespace, object);

 
      create: function(name, value) {
        var node = window.razor,
        nameParts = name ? name.split('.') : [],
        c = nameParts.length;
        for (var i = 0; i < c; i++) {
          var part = nameParts[i];
          var nso = node[part];
          if (!nso) {
            nso = (value && i + 1 === c) ? value : {};
            node[part] = nso;
          }
          node = nso;
        }
      return node;
    },
